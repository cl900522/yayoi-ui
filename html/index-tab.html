UI页签测试

可以创建页签，以及增加删除页签

        
    $(function(){
        var tab = new yayoi.ui.tab.Tab({
            selector: "#tabId",
            tabs: [
               {code: 1, title: "表单测试", url: "./index-form.html", closeable: true},
               {code: 2, title: "模型测试", url: "./index-model.html", closeable: true},
               {code: 3, title: "基础模块测试", url: "./index-core.html", closeable: true},
               {code: 4, title: "百度", url: "https://www.baidu.com", closeable: false},
            ]
        });
        tab.addTab({code: 5, title: "京东", url: "http://www.jingdong.com", closeable: false});
        tab.addTab({code: 6, title: "HTML页签", html: "HTML TAB TEST", closeable: false});
    })
