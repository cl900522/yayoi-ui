UI对话框测试

可以创建对话框，并绑定函数

        
            var dialog = new yayoi.ui.window.Dialog();
            function showDialog() {
                dialog.show("成功", "数据提交成功", function(){
                    yayoi.log.info("Function callback success.")
                });
            }
            function hideDialog() {
                dialog.close();
            }
