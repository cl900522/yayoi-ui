YayoiUI核心测试

extend函数，用于辅助继承类

        
    var People = yayoi.util.extend("yayoi.common.People", "Object", [], function() {
        this.name = null;
        this.age = null;
        this.getAge = function() {
            return this.age;
        }
    });

path.Router路径帮助函数

log.Logger用于帮助输出日志

        
    var path = new yayoi.ui.path.Router("./../tmp/tmp/google");
    yayoi.log.info(path.pwd());
    path.cd("../httpd");

    yayoi.log.info(path.pwd());
