UI菜单

可以创建菜单

        
        var menu = new yayoi.ui.menu.Menu({
            nodes: [
                {
                    text: "Open",
                    icon: "file-alt",
                    click: function() {
                        yayoi.log.info("Very well, menu clicked.")
                    }
                },
                {
                    text: "Open Recent Files",
                    icon: "book",
                    click: function() {
                        yayoi.log.info("Very well, menu clicked.")
                    },
                    subMenu: {
                        nodes: [
                            {
                                text: "index.html",
                            },
                            {
                                text: "Disabled",
                                icon: "lock",
                                click: function() {
                                    yayoi.log.info("Very well, menu clicked.")
                                },
                                disabled: true
                            }
                        ]
                    }
                }
            ]
        });
